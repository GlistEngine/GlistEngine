35 class gAnimationTrigger {
36 public:
37  template<typename First, typename Second>
38  constexpr gAnimationTrigger(First* first, Second* second, TriggerCondition cond)
39  {
40  switch(cond)
41  {
42  case TriggerCondition::LESS:
43  triggerfunc = [first, second]() -> bool { return *first < *second; };
44  break;
45  case TriggerCondition::GREATER:
46  triggerfunc = [first, second]() -> bool { return *first > *second; };
47  break;
48  case TriggerCondition::LESS_EQUAL:
49  triggerfunc = [first, second]() -> bool { return *first <= *second; };
50  break;
51  case TriggerCondition::GREATER_EQUAL:
52  triggerfunc = [first, second]() -> bool { return *first >= *second; };
53  break;
54  case TriggerCondition::EQUAL:
55  triggerfunc = [first, second]() -> bool { return *first == *second; };
56  break;
57  case TriggerCondition::FALSE:
58  triggerfunc = [first]() -> bool { return *first == false; };
59  break;
60  case TriggerCondition::TRUE:
61  triggerfunc = [first]() -> bool { return *first == true; };
62  break;
63  }
64  }
65 
66  bool isConditionTriggered() {
67  return triggerfunc();
68  }
69 private:
70  std::function<bool()> triggerfunc;
71 };
